import { Injectable,Controller,
    Delete,
    Get,
    Post,
    Put,
    Status } from "@vigilio/express-core";


@Injectable()
@Controller("/${props}")
export class ${Props}ApiController {
    constructor() {}

    @Get("/")
    async index() {}

    @Get("/:id")
    async show() {}

    @Status(201)
    @Post("/")
    async store() {}

    @Status(200)
    @Put("/")
    async update() {}

    @Status(204)
    @Delete("/")
    async destroy() {}
}
