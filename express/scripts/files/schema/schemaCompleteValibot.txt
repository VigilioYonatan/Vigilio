import { number, minLength, Input, string,transform, object } from "valibot";
import { slug } from "@vigilio/express-core/helpers";

export const ${props}Schema = object({
    id: number(),
    name: string("Este campo es obligatorio", [minLength(1,"Este campo es obligatorio"),minLength(3,"Este campo requiere mínimo 3 carácteres")]),
    slug: transform(string("Este campo es obligatorio", [minLength(1,"Este campo es obligatorio"),minLength(3,"Este campo requiere mínimo 3 carácteres")]),slug),
});

export type ${Props}Schema = Input<typeof ${props}Schema>;
export type ${Props}EntitySchema = Omit<${Props}Schema, "id">;
